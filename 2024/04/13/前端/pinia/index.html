pinia

前端
发布时间 : 2024-04-13 14:21
阅读 :
  1. 在src下面创建一个store文件夹 创建index.ts
  2. 在需要用到pinia的vue组件里引用

在src下面创建一个store文件夹 创建index.ts

import { ref, computed } from ‘vue’
import { defineStore } from ‘pinia’

export const useMain = defineStore(‘main’, () => {
const counter = ref(0)
const name = ref(‘Eduardo’)

const doubleCount = computed(() => counter.value * 2)

function increment() {
    counter.value++
}

function randomizeCounter() {
    setTimeout(() => {
        counter.value = Math.round(100 * Math.random())
    }, 0);
}
return { counter, name, doubleCount, increment, randomizeCounter }

})

在需要用到pinia的vue组件里引用

import { useMain } from “../store/index”;

然后定义
const CounterStore = useMain();

然后就可以直接使用
console.log(CounterStore.counter)
