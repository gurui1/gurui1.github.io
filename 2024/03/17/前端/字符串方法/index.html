字符串方法

前端
发布时间 : 2024-03-17 16:04
阅读 :
  1. concat()：用于连接两个或多个字符串，并返回新的字符串。
  2. includes()：用于判断一个字符串中是否包含指定的子字符串，返回 true 或 false。
  3. indexOf()：用于返回一个字符串在另一个字符串中第一次出现的位置，如果没有找到，则返回 - 1。
  4. charAt()：用于返回指定索引位置的字符。
  5. split()：用于将一个字符串分割成子字符串并将结果作为字符串数组返回。
  6. toUpperCase()：用于将字符串转换为大写。
  7. toLowerCase()：用于将字符串转换为小写。
  8. slice()：用于提取字符串的片段，并返回一个新的字符串。
  9. substring()：类似于 slice()，但不接受负的索引。
  10. replace()：用于在字符串中替换指定子字符串。

concat()：用于连接两个或多个字符串，并返回新的字符串。

let str1 = “Hello”;
let str2 = “World”;
let newStr = str1.concat(“”, str2);
console.log(newStr); ## 输出：Hello World

includes()：用于判断一个字符串中是否包含指定的子字符串，返回 true 或 false。

let str = “Hello, World!”;
let isIncludes = str.includes(“World”);
console.log(isIncludes); ## 输出：true

indexOf()：用于返回一个字符串在另一个字符串中第一次出现的位置，如果没有找到，则返回 - 1。

let str4 = “Hello, World!”;
let index = str4.indexOf(“World”);
console.log(index); ## 输出：7

charAt()：用于返回指定索引位置的字符。

let str5 = “Hello, World!”;
let char = str5.charAt(1);
console.log(char); ## 输出：e

split()：用于将一个字符串分割成子字符串并将结果作为字符串数组返回。

let str6 = “apple,banana,orange”;
let arr = str6.split(“,”);## 以“”内部的为分节符
console.log(arr); ## 输出：[“apple”, “banana”, “orange”]

toUpperCase()：用于将字符串转换为大写。

toLowerCase()：用于将字符串转换为小写。

let str7 = “Hello, World!”;
let upperCaseStr = str7.toUpperCase();
let lowerCaseStr = str7.toLowerCase();
console.log(upperCaseStr); ## 输出：HELLO, WORLD!
console.log(lowerCaseStr); ## 输出：hello, world!

slice()：用于提取字符串的片段，并返回一个新的字符串。

substring()：类似于 slice()，但不接受负的索引。

let str8 = “Hello,World!”;
let subStr = str8.slice(6);## 从（）里这一位开始提取
console.log(subStr); ## 输出：World!

replace()：用于在字符串中替换指定子字符串。

let str9= “Hello, World!”;
let newStr1 = str9.replace(“World”, “JavaScript”);
console.log(newStr1); ## 输出：Hello, JavaScript!
